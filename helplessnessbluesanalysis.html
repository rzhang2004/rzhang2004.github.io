Import libraries

import spotipy
from spotipy.oauth2 import SpotifyClientCredentials
import matplotlib.pyplot as plt
import pandas as pd
import seaborn as sns
from scipy import stats

with open("credentials.txt") as f:
    credentials = f.read().splitlines()

client_id = str(credentials[0])
client_secret = str(credentials[1])

# Authenticate
client_credentials_manager = SpotifyClientCredentials(client_id=client_id, client_secret=client_secret)
sp = spotipy.Spotify(client_credentials_manager=client_credentials_manager)

Extract basic album information

album_id = "7D0rCfJjFj9x0bdgRKtvzb"  # Album id for Fleet Foxes' "Helplessness Blues" album

album = sp.album(album_id)

# Extract basic album info
print(f"Album Name: {album['name']}")
print(f"Artist: {album['artists'][0]['name']}")
print(f"Release Date: {album['release_date']}")
print(f"Popularity: {album['popularity']}/100")

track_names = []
popularity_scores = []
tracks = sp.album_tracks(album_id)["items"]

for track in tracks:
    track_data = sp.track(track["id"])
    track_names.append(track_data["name"])
    popularity_scores.append(track_data["popularity"])

# Plot
plt.figure(figsize=(8, 4))
plt.bar(track_names, popularity_scores, color='skyblue')
plt.xlabel("Tracks")
plt.ylabel("Popularity (0-100)")
plt.title("Track Popularity in Album")
plt.xticks(rotation=45, ha='right')
plt.tight_layout()
plt.show()

Perform t-test

odd_song_scores = info.loc[info['odd'] == True, 'popularity']
normal_song_scores = info.loc[info['odd'] == False, 'popularity']

# Welch’s t-test (doesn’t assume equal variances, safer for small samples)
t_stat, p_val = stats.ttest_ind(odd_song_scores, normal_song_scores, equal_var=False)

print(f"T-statistic: {t_stat:.4f}, P-value: {p_val:.4f}")
T-statistic: 1.7394, P-value: 0.1332

Summarization

The t-test yielded a p-value of 0.1332, indicating no statistically significant difference in average popularity between songs with nontraditional sectional lengths and those with traditional lengths. However, the sample size is relatively small, and it is noteworthy that the three most popular tracks on Helplessness Blues all employ nontraditional sectional lengths.
