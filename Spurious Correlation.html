import requests
import pandas as pd
from tqdm import tqdm

def get_all_games(username):
    username = username.lower()  # API usernames are lowercase only
    base_url = f"https://api.chess.com/pub/player/{username}"
    
    # Step 1: Get all archive URLs
    archives_url = f"{base_url}/games/archives"
    headers = {"User-Agent": "Bobby"}  # avoid 403 errors
    resp = requests.get(archives_url, headers=headers)
    
    if resp.status_code != 200:
        raise Exception(f"Error {resp.status_code}: {resp.text}")
    
    archives = resp.json().get("archives", [])
    all_games = []
    
    # Step 2: Loop through archives
    for url in tqdm(archives, desc="Fetching archives"):
        data = requests.get(url, headers=headers).json()
        games = data.get("games", [])
        all_games.extend(games)
    
    return all_games

def fetch_pageviews(topic, start="20230801", end="20251001"):
    url = f"https://wikimedia.org/api/rest_v1/metrics/pageviews/per-article/en.wikipedia/all-access/all-agents/{topic}/monthly/{start}/{end}"
    headers = {'User-Agent': 'rayzhang2027@u.northwestern.edu'}
    r = requests.get(url, headers=headers)
    if r.status_code != 200:
        print(f"Error {r.status_code} for {topic}")
        return pd.Series(dtype="float64")
    data = r.json()["items"]
    df = pd.DataFrame([{
        "date": item["timestamp"][:6],
        "views": item["views"]
    } for item in data])
    df["date"] = pd.to_datetime(df["date"], format="%Y%m")
    return df.set_index("date")["views"]
