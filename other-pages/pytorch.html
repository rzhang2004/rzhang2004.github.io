Customize prompt

def job_prompt(jobs):

    '''
    Defines a custom prompt for some LLM to receive a list of job titles, and then
    output a Python dictionary where each key is a job title and the values are the
    Q1, median, and Q3 incomes for that job in the US. Note: US is being used because
    it has more recent and reliable data from the US Census Bureau.
    The input, jobs, should be a pandas Series such as X['job'].unique()
    '''
    
    prompt = f'''
    Given the following list of job titles, return a Python dictionary where each key is a job title (as a string) 
    and the value is a list of three numbers representing the 25th percentile income, the median income, 
    and the 75th percentile income for that job in the US. Format the dictionary as follows:
    
    {{
        "job title 1": [25th_percentile_income, median_income, 75th_percentile_income],
        "job title 2": [25th_percentile_income, median_income, 75th_percentile_income],
        ...
    }}
    
    All incomes should be in USD and rounded to the nearest whole number. Use the most recent and reliable publicly available data (e.g., U.S. Bureau of Labor Statistics).
    Make sure to return information for each and every job on the list. Here is the list of job titles:
    
    {jobs}
    '''
    return prompt

Send prompt to LLM

def send_prompt(prompt):

    '''
    Returns only text response to given prompt.
    '''
    
    response = openai.ChatCompletion.create(
        model="deepseek-ai/DeepSeek-R1-Distill-Llama-70B:novita",
        messages=[
            {"role": "user", "content": prompt}
        ]
    )

    return response.choices[0].message.content

Extract dictionary from dictionary-like string

def extract_and_parse_dict(response_text):
    """
    Extracts a Python-like dictionary from a larger text blob and converts it to an actual dict.

    Parameters:
        response_text (str): The full string containing a code block with a dictionary.

    Returns:
        dict: Parsed dictionary object.
    """
    # Try to extract text inside a ```python code block
    match = re.search(r"```(?:python)?\s*(\{.*?\})\s*```", response_text, re.DOTALL)
    
    # If no code block found, try to extract dictionary-like structure manually
    if not match:
        match = re.search(r"(\{.*?\})", response_text, re.DOTALL)

    if not match:
        raise ValueError("No dictionary found in response.")

    dict_str = match.group(1)

    try:
        return ast.literal_eval(dict_str)
    except Exception as e:
        raise ValueError(f"Failed to parse dictionary: {e}")

Define model architecture

import torch.nn as nn
import torch

class BankModelRes(nn.Module):
    def __init__(self, embedding_info, num_numerical, output_dim, hidden_dim=256, dropout=0.2):
        super().__init__()

        # Embedding layers
        self.embeddings = nn.ModuleList([
            nn.Embedding(num_categories, emb_dim)
            for num_categories, emb_dim in embedding_info
        ])
        total_emb_dim = sum(emb_dim for _, emb_dim in embedding_info)

        input_dim = total_emb_dim + num_numerical

        # Main network
        self.input_layer = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.BatchNorm1d(hidden_dim),
            nn.SiLU()
        )
        self.res_blocks = nn.Sequential(
            ResidualBlock(hidden_dim, dropout),
            ResidualBlock(hidden_dim, dropout),
        )
        self.output_layer = nn.Sequential(
            nn.Dropout(dropout),
            nn.Linear(hidden_dim, output_dim)
        )

    def forward(self, x_cat, x_num):
        # Embed each categorical column and concatenate
        embedded = [emb(x_cat[:, i]) for i, emb in enumerate(self.embeddings)]
        x = torch.cat(embedded + [x_num], dim=1)

        x = self.input_layer(x)
        x = self.res_blocks(x)
        return self.output_layer(x)

class ResidualBlock(nn.Module):
    def __init__(self, dim, dropout=0.2):
        super().__init__()
        self.block = nn.Sequential(
            nn.Linear(dim, dim),
            nn.BatchNorm1d(dim),
            nn.SiLU(),
            nn.Dropout(dropout),
            nn.Linear(dim, dim),
            nn.BatchNorm1d(dim),
            nn.SiLU(),
            nn.Dropout(dropout),
        )

    def forward(self, x):
        return x + self.block(x)

Helper function for embedding info

def get_embedding_info(df, categorical_cols):
    embedding_info = []
    for col in categorical_cols:
        num_categories = df[col].nunique()
        emb_dim = min(50, (num_categories + 1) // 2)
        embedding_info.append((num_categories, emb_dim))
    return embedding_info

Training loop

X_train_cat, X_train_num, y_train_tensor = X_train_cat.to(device), X_train_num.to(device), y_train_tensor.to(device)
X_val_cat, X_val_num, y_val_tensor = X_val_cat.to(device), X_val_num.to(device), y_val_tensor.to(device)
batch_size = 2048  # Adjust based on your GPU memory

train_dataset = TensorDataset(X_train_cat, X_train_num, y_train_tensor)
val_dataset = TensorDataset(X_val_cat, X_val_num, y_val_tensor)

train_loader = DataLoader(train_dataset, batch_size=batch_size, shuffle=True)
val_loader = DataLoader(val_dataset, batch_size=batch_size, shuffle=False)
epochs = 1000
patience = 40
train_roc = []
valid_roc = []
best_val_roc = 0

for epoch in range(epochs):
    model.train()
    epoch_preds = []
    epoch_labels = []

    for X_cat_batch, X_num_batch, y_batch in train_loader:
        optimizer.zero_grad()

        logits = model(X_cat_batch, X_num_batch).squeeze()
        pred_probs = torch.sigmoid(logits)
        preds = torch.round(pred_probs)

        loss = loss_fn(logits, y_batch)

        loss.backward()
        optimizer.step()

        epoch_preds.append(pred_probs.detach().cpu())
        epoch_labels.append(y_batch.detach().cpu())

    # Compute ROC AUC for epoch
    epoch_preds = torch.cat(epoch_preds).numpy()
    epoch_labels = torch.cat(epoch_labels).numpy()
    roc = roc_auc_score(epoch_labels, epoch_preds)
    train_roc.append(roc)

    # Validation
    model.eval()
    val_preds = []
    val_labels = []

    with torch.inference_mode():
        for X_cat_batch, X_num_batch, y_batch in val_loader:
            logits = model(X_cat_batch, X_num_batch).squeeze()
            probs = torch.sigmoid(logits)

            val_preds.append(probs.cpu())
            val_labels.append(y_batch.cpu())

    val_preds = torch.cat(val_preds).numpy()
    val_labels = torch.cat(val_labels).numpy()
    val_roc = roc_auc_score(val_labels, val_preds)
    valid_roc.append(val_roc)

    if epoch % 20 == 0:
        print(f'Epoch: {epoch} | Validation ROC: {val_roc:.4f}')

    # Early stopping
    if val_roc > best_val_roc:
        best_val_roc = val_roc
        early_stop_counter = 0
        best_model_state = model.state_dict()
    else:
        early_stop_counter += 1
        if early_stop_counter >= patience:
            print(f"Early stopping at epoch {epoch+1}. Best Val ROC AUC: {best_val_roc:.4f}")
            last_epoch = epoch + 1
            break
